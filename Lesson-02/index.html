TypeScript Basic Lesson - 02


        
            

                

                    By Inference/Implicit
 
                    let userName = "Saquib";

                    console.log(userName);

                    // userName type 'string'
                    


                    const num1 = 10;

                    console.log(num1);

                    // num1 type 'number'
                

                

                    By Explicit 

                    const anotherUserName: string = "Asghar"

                    console.log(anotherUserName);

                    // anotherUserName type 'string'
                    


                    const num2: number = 234;

                    console.log(num2);

                    // num2 type 'number'
                

                

                    Error 

                    userName = 29;

                    //Type 'number' is not assignable to type 'string'.
                    


                    anotherUserName = true 

                    //Cannot assign to 'anotherUserName' because it is a constant.
